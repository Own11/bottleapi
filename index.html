API Интеграция

example-request.py 
import requests

url = "https://api.lumenlabs.kz/v1/chat"
payload = {
    "model": "bottle-1-nano",
    "message": "Объясни закон Ома для LabAI"
}

response = requests.post(url, json=payload, headers=headers)
print(response.json())
